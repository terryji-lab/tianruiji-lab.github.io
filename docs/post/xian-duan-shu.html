线段树

线段树

线段是是一种二叉树，用于高效动态的维护数组的区间信息（最值，和，gcd等），可以以O(logn)的复杂度进行单点/区间的修改与查询。

数据结构

struct node 
{
    int l, r;        // 当前节点管理的区间 [l, r]
    int lazy;        // 懒惰标记（本节点已更新，但子节点还未下传的加法值）
    int sum;         // 区间的总和
} tr[4 * N];         // 开 4 倍数组大小
int w[N];            // 原始数组

函数

左右子节点编号

使用数组来存储树，那么一个节点p的左儿子下标为2*p，右儿子下标为2*p+1。

int lc(int p) { return p * 2; }      // left child
int rc(int p) { return p * 2 + 1; }  // right child

pushup向上更新

void pushup(int p) 
{
    tr[p].sum = tr[lc(p)].sum + tr[rc(p)].sum;
}

当左右儿子更新后，用它们的 sum 更新当前节点的 sum。在修改或建树回溯时调用。

pushdown下传懒标记

void pushdown(int p) 
{
    if (tr[p].lazy) {
        tr[lc(p)].sum += tr[p].lazy * (tr[lc(p)].r - tr[lc(p)].l + 1);
        tr[rc(p)].sum += tr[p].lazy * (tr[rc(p)].r - tr[rc(p)].l + 1);
        tr[lc(p)].lazy += tr[p].lazy;
        tr[rc(p)].lazy += tr[p].lazy;
        tr[p].lazy = 0;
    }
}
  • 将当前节点的懒标记下推到左右儿子。
  • 更新儿子的 sum（懒标记值 × 儿子区间长度）。
  • 累加儿子的 lazy。
  • 清空当前节点的 lazy。
    在访问子节点之前（比如更新或查询递归进入下一层时）就需要调用这个函数。

build建树

void build(int p, int l, int r) 
{
    tr[p].l = l;
    tr[p].r = r;
    tr[p].sum = w[l];          // 先初始化，叶子节点值确定后再 pushup 会修正非叶子节点
    if (l == r) return;       // 叶子节点直接返回
    int m = (l + r) / 2;
    build(lc(p), l, m);
    build(rc(p), m + 1, r);
    pushup(p);
}

先递归到叶子节点赋上原数组的值，然后所有非叶子节点都会调用pushup反向传播合成自己的值。

update区间修改

void update(int p, int x, int y, int k) 
{
    // 当前区间完全在目标区间内 → 直接更新并打懒标记
    if (tr[p].l >= x && tr[p].r <= y) {
        tr[p].sum += k * (tr[p].r - tr[p].l + 1);
        tr[p].lazy += k;
        return;
    }
    int m = (tr[p].l + tr[p].r) / 2;
    pushdown(p);                          // 下传旧标记，确保儿子信息正确
    if (x <= m) update(lc(p), x, y, k);   // 左子区间有交集
    if (y > m)  update(rc(p), x, y, k);   // 右子区间有交集
    pushup(p);                            // 用儿子更新当前节点
}
  • 当前区间 [l,r] 完全被 [x,y] 覆盖 → 直接加并打懒标记，结束。
  • 否则，先 pushdown 下传标记。
  • 递归处理左右有交集的部分。
  • 回溯时 pushup 维护准确性。
  • 同时也可以用于单点修改当参数中的x==y时

query区间查询

int query(int p, int x, int y) 
{
    if (x <= tr[p].l && y >= tr[p].r)       // 当前区间完全包含在查询区间内
        return tr[p].sum;
    int m = (tr[p].l + tr[p].r) / 2;
    pushdown(p);                            // 查询前也要下传标记
    int sum = 0;
    if (x <= m) sum += query(lc(p), x, y);
    if (y > m)  sum += query(rc(p), x, y);
    return sum;
}

查询同样遵循递归拆分区间，区间完全覆盖则直接返回 sum，否则 pushdown 后进入儿子。

完整代码

#include <bits/stdc++.h>
#define int long long          // 防止和 / 懒标记溢出
using namespace std;

int n, m;
const int N = 5e5 + 5;

struct node {
    int l, r, lazy, sum;
} tr[4 * N];
int w[N];

int lc(int p) { return p * 2; }
int rc(int p) { return p * 2 + 1; }

void pushup(int p) {
    tr[p].sum = tr[lc(p)].sum + tr[rc(p)].sum;
}

void pushdown(int p) {
    if (tr[p].lazy) {
        tr[lc(p)].sum += tr[p].lazy * (tr[lc(p)].r - tr[lc(p)].l + 1);
        tr[rc(p)].sum += tr[p].lazy * (tr[rc(p)].r - tr[rc(p)].l + 1);
        tr[lc(p)].lazy += tr[p].lazy;
        tr[rc(p)].lazy += tr[p].lazy;
        tr[p].lazy = 0;
    }
}

void build(int p, int l, int r) {
    tr[p].l = l;
    tr[p].r = r;
    tr[p].sum = w[l];            // 先赋初值，叶子节点会直接返回，非叶子被 pushup 覆盖
    if (l == r) return;
    int m = (l + r) / 2;
    build(lc(p), l, m);
    build(rc(p), m + 1, r);
    pushup(p);
}

void update(int p, int x, int y, int k) {
    if (tr[p].l >= x && tr[p].r <= y) {
        tr[p].sum += k * (tr[p].r - tr[p].l + 1);
        tr[p].lazy += k;
        return;
    }
    int m = (tr[p].l + tr[p].r) / 2;
    pushdown(p);
    if (x <= m) update(lc(p), x, y, k);
    if (y > m)  update(rc(p), x, y, k);
    pushup(p);
}

int query(int p, int x, int y) {
    if (x <= tr[p].l && y >= tr[p].r)
        return tr[p].sum;
    int m = (tr[p].l + tr[p].r) / 2;
    pushdown(p);
    int sum = 0;
    if (x <= m) sum += query(lc(p), x, y);
    if (y > m)  sum += query(rc(p), x, y);
    return sum;
}

signed main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> w[i];
    build(1, 1, n);
    while (m--) {
        int op, x, y, k;
        cin >> op;
        if (op == 1) {
            cin >> x >> y >> k;
            update(1, x, y, k);
        } else {
            cin >> x >> y;
            cout << query(1, x, y) << '\n';
        }
    }
    return 0;
}

线段树的难点在于懒惰标记的管理。记住一条原则：“用到儿子时，必先 pushdown”，并且回溯时及时 pushup，代码就不容易出错。
