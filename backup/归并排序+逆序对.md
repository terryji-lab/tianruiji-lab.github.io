# 归并排序
和快速排序一样，归并排序也使用了分治的思想。不同的是，快速排序是先在大区间进行分类，再在小区间进行分类，直到没有更小的区间。而归并排序是先让小区间有序，然后再合并有序的小区间。一个是自上而下的，一个是自下而上的。归并排序的优点在于我们已经知道两个有序数列合并的时间复杂度为O(n)，可以大大降低时间复杂度，并且是一个稳定的排序。
## 代码实现
```cpp
#include <bits/stdc++.h>
using namespace std;
const int N=1e5;
int a[N],b[N],n;
void msort(int l,int r)
{
    if(l==r)return;//最小区间只有一个数字，已经是有序区间，直接返回
    int mid=l+r>>1;
    msort(l,mid);
    msort(mid+1,r);//递归到最小区间

    int i=l,j=mid+1,k=l;//l是左区间起点，mid是左区间终点，mid+1是右区间起点，r是右区间终点
    while(i<=mid && j<=r)
    {
        if(a[i]<=a[j])b[k++]=a[i++];
        else b[k++]=a[j++];//区间合并
    }
    while(i<=mid)b[k++]=a[i++];
    while(j<=r)b[k++]=a[j++];//总有一个区间的元素先被选完，所以清理剩余元素，得到有序区间b
    for(int i=l;i<=r;i++)a[i]=b[i];//将b复制给a，让待排序数组a的区间[l,r]有序
}
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)cin>>a[i];
    msort(1,n);
    for(int i=1;i<=n;i++)cout<<a[i]<<' ';
    return 0;
}
```
